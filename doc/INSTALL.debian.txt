The following is command example of installing HiSocial to a new Debian/Ubuntu server.

The json cgi will be installed to http://domain.com/json-cgi/json_cmd.py .
The json web console will be installed to http://domain.com/json-console/ .
The web client is not available yet.

# : Root shell commands
$ : Normal user shell commands
> : Program shell commands

==========================================

# apt-get install git lighttpd mysql-server python-crypto python-sqlalchemy python-mysqldb python-nose python-recaptcha -y

# mkdir /HISOCIAL_WWW
# chmod -R /HISOCIAL_WWW USERNAME:USERNAME # create directory for installation

$ mysql --user=root mysql --password=MYSQL_ROOT_PASSWORD
> CREATE USER 'MYSQL_USERNAME'@'localhost' IDENTIFIED BY 'MYSQL_PASSWORD';
> CREATE DATABASE MYSQL_DATABASE;
> GRANT ALL ON MYSQL_DATABASE.* TO 'MYSQL_USERNAME'@'localhost';
> quit

$ mkdir /WORKING_DIR
$ cd /WORKING_DIR
$ git clone git://github.com/luzi82/HiSocial.git -b test-server
$ cd /WORKING_DIR/HiSocial/tool
$ cp /WORKING_DIR/HiSocial/tool/install_config.py.hiserver.example /WORKING_DIR/HiSocial/tool/install_config.py 
$ nano /WORKING_DIR/HiSocial/tool/install_config.py # Modify it

$ cd /WORKING_DIR/HiSocial/tool/
$ ./install.py

# nano /etc/lighttpd/conf-available/99-hiauntie.conf

### /etc/lighttpd/conf-available/99-hiauntie.conf ###

static-file.exclude-extensions += ( ".py " )

alias.url += ( "/json-cgi" => "/HISOCIAL_WWW/json-cgi" )
$HTTP["url"] =~ "^/json-cgi/" {
	cgi.assign += ( ".py" => "/usr/bin/python" )
	dir-listing.activate = "disable"
}
$HTTP["url"] =~ "^/json-cgi/webjsoncgi_config.py$" { url.access-deny = ("") }

alias.url += ( "/json-console" => "/HISOCIAL_WWW/json-console" )
$HTTP["url"] =~ "^/json-console/" {
	dir-listing.activate = "disable"
}
